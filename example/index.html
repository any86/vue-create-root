单例1,插入到body的头部

            
                const C = Vue.createRootClass(Com, {
                    insertPosition: 'prepend'
                });
        
                Vue.prototype.$single = (props) => C.init({
                    isShow: true,
                    ...props
                });

                // 关闭按钮逻辑
                Vue.prototype.$single.close = () => {
                    C.$update({
                        isShow: false
                    })
                }
